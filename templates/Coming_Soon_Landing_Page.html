HTML Structure 
<div class="load" id="loadmodalbox">
  <div class="load-modal-box">
    <div class="sandtimer-icon">⏳</div>
    <h2 class="loading-text">Coming Soon...<br>Try something else in the meantime</h2>
  </div>
</div>
CSS Animation 
.load-modal-box {
  border-radius: 8px;
  -webkit-box-shadow: 0px 1px 1px 0px rgba(0, 0, 0, 0.2);
  -moz-box-shadow: 0px 1px 1px 0px rgba(0, 0, 0, 0.2);
  box-shadow: 0px 1px 1px 0px rgba(0, 0, 0, 0.2);
  width: 400px;
  height: 300px;
  background: -webkit-linear-gradient(top, #f1f1f1 65%, #f1f1f1 65%);
  background: linear-gradient(to bottom, #f1f1f1 65%, #f1f1f1 65%);
  z-index: 3;
  position: relative;
  overflow: hidden;
  margin: 0 auto;
  display: flex;
  flex-direction: column;
  align-items: center;
  justify-content: center;
}

.load {
  position: relative;
  display: flex;
  justify-content: center;
  align-items: center;
}

.sandtimer-icon {
  font-size: 5rem;
  position: absolute;
  top: 80px;
  animation: rotateHourglass 2s infinite ease-in-out;
  filter: drop-shadow(0 4px 8px rgba(0, 0, 0, 0.3));
}

.loading-text {
  text-align: center;
  color: #6f7475;
  position: absolute;
  top: 200px;
  font-size: 1.2rem;
  font-weight: 500;
  margin: 0;
  animation: pulse 1.5s infinite ease-in-out;
}

@keyframes rotateHourglass {
  0% { 
    transform: rotate(0deg); 
    opacity: 1; 
  }
  25% { 
    transform: rotate(0deg); 
    opacity: 0.8; 
  }
  50% { 
    transform: rotate(180deg); 
    opacity: 1; 
  }
  75% { 
    transform: rotate(180deg); 
    opacity: 0.8; 
  }
  100% { 
    transform: rotate(360deg); 
    opacity: 1; 
  }
}

@keyframes pulse {
  0%, 100% { 
    opacity: 1; 
    transform: scale(1); 
  }
  50% { 
    opacity: 0.7; 
    transform: scale(0.98); 
  }
}

/* Responsive Design */
@media (max-width: 768px) {
  .load-modal-box {
    width: 300px;
    height: 250px;
  }
  
  .sandtimer-icon {
    font-size: 4rem;
    top: 60px;
  }
  
  .loading-text {
    top: 160px;
    font-size: 1rem;
  }
}
